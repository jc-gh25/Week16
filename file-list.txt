# Container configuration and orchestration
Dockerfile
  - Multi-stage build: Maven build stage + Alpine runtime stage
  - Includes AWS CLI and jq for Route 53 DNS updates
  - Route 53 update script: /update-route53.sh
  - Startup script: /startup.sh
  - Hosted Zone ID: Z08164103KW73VKOVN6GY
  - Domain: project.jcarl.net
  - TTL: 300 seconds
  - Automatic DNS update on container startup

docker-compose.yaml
  - MySQL 8.0 service configuration
  - Spring Boot application service
  - Health checks and volume management
  - Environment variables for database connection

# SOURCE CODE - MAIN APPLICATION
# ============================================================================

# Application Entry Point
src/main/java/music/library/MusicLibraryApplication.java

# Configuration Classes
src/main/java/music/library/config/
  CorsConfig.java - CORS configuration for cross-origin requests
  PageConfig.java - Pagination configuration
  SwaggerConfig.java - Swagger/OpenAPI documentation configuration

# Controllers
src/main/java/music/library/controller/
  MusicLibraryController.java - Main REST API endpoints
  SwaggerRedirectController.java - Swagger UI redirect handler
  error/ - Error handling controllers
  model/ - Controller model classes

# Data Transfer Objects (DTOs)
src/main/java/music/library/dto/
  ApiInfoResponse.java - API information response
  CreateAlbumRequest.java - Album creation request
  CreateArtistRequest.java - Artist creation request
  CreateGenreRequest.java - Genre creation request
  DatabaseResetResponse.java - Database reset response
  UpdateAlbumRequest.java - Album update request
  UpdateArtistRequest.java - Artist update request
  UpdateGenreRequest.java - Genre update request

# Entity Classes (JPA/Hibernate)
src/main/java/music/library/entity/
  Album.java - Album entity with relationships
  Artist.java - Artist entity with relationships
  Genre.java - Genre entity with relationships

# Exception Handling
src/main/java/music/library/exception/
  ApiError.java - API error response model
  GlobalExceptionHandler.java - Global exception handler
  ResourceNotFoundException.java - Resource not found exception

# Repository Layer (Data Access)
src/main/java/music/library/repository/
  AlbumRepository.java - Album data access
  ArtistRepository.java - Artist data access
  GenreRepository.java - Genre data access

# Service Layer (Business Logic)
src/main/java/music/library/service/
  AlbumService.java - Album business logic
  ArtistService.java - Artist business logic
  DatabaseResetService.java - Database reset functionality
  GenreService.java - Genre business logic

# Specifications (JPA Criteria API)
src/main/java/music/library/specification/
  AlbumSpecs.java - Album search specifications

src/main/resources/application.properties

  - MySQL database connection settings

  - SQL initialization mode

  V1__Create_Schema.sql - Initial schema creation
  V1__Initial_Schema.sql.sql - Alternative schema file
  V1_1__add_catalog_number_unique.sql - Add unique catalog number constraint
  V1_2__add_cover_image_url.sql - Add cover image URL column
  V2__Seed_Data.sql - Initial data seeding

# PROJECT SUMMARY
# ============================================================================
#
# Technology Stack:
#   - Java 17
#   - Spring Boot 3.5.7
#   - Spring Data JPA / Hibernate
#   - MySQL 8.0
#   - Flyway (Database migrations)
#   - Swagger/OpenAPI (API documentation)
#   - Docker & Docker Compose
#   - AWS ECS Fargate (Container orchestration)
#   - AWS Route 53 (DNS management)
#
# Key Features:
#   - RESTful API for music library management
#   - Album, Artist, and Genre management
#   - Database reset functionality
#   - CORS support for cross-origin requests
#   - Swagger UI for API documentation
#   - Automatic Route 53 DNS updates on container startup
#   - Pagination support for large datasets
#   - Comprehensive error handling
#   - Web UI for browsing music library
#
# Deployment:
#   - Containerized with Docker
#   - Multi-stage build for optimized image size
#   - AWS ECS Fargate deployment
#   - Route 53 DNS configuration (project.jcarl.net)
#   - MySQL database on AWS RDS
#
